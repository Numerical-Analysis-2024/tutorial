Sustitución hacia adelante#

def forsub(L, bs):
    """
    n: size de x
    """
    n = bs.size
    xs = np.zeros(n)
    for i in range(n):
        # i recorre desde 0 hasta n-1
        xs[i] = (bs[i] - L[i, :i] @ xs[:i]) / L[i, i]
    return xs

Sustitución hacia atrás#

def backsub(U, bs):
    n = bs.size
    xs = np.zeros(n)
    for i in reversed(range(n)):
        xs[i] = (bs[i] - U[i, i + 1 :] @ xs[i + 1 :]) / U[i, i]
    return xs

Eliminación Gaussiana#

def gauelim(inA, inbs):
    A = np.copy(inA)
    bs = np.copy(inbs)
    n = bs.size

    for j in range(n - 1):
        for i in range(j + 1, n):
            coeff = A[i, j] / A[j, j]
            A[i, j:] -= coeff * A[j, j:]
            bs[i] -= coeff * bs[j]

    xs = backsub(A, bs)
    return xs

Descomposición LU#

def ludec(A):
    n = A.shape[0]
    U = np.copy(A)
    L = np.identity(n)

    for j in range(n - 1):
        for i in range(j + 1, n):
            coeff = U[i, j] / U[j, j]
            U[i, j:] -= coeff * U[j, j:]
            L[i, j] = coeff

    return L, U

Resolución de un sistema lineal vía LU#

def lusolve(A, bs):
    L, U = ludec(A)
    ys = forsub(L, bs)
    xs = backsub(U, ys)
    return xs
